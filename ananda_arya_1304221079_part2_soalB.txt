program luasVolumeKubus
{ program menghitung nilai volume dan luas kubus}
kamus
  sisi : integer
  volume : integer
  luas : integer

algoritma
  input(sisi)
  volume <- sisi * sisi * sisi
  luas <- 6 * sisi * sisi
  output(volume, luas)
endprogram

program binnerConvertion
{ mengkoversi bilangan binner ke desimal}
  kamus
    a : integer
    b : integer
    c : integer
    d : integer
    hasil : integer

algoritma
  input(a, b, c, d)
  hasil <- a * 8 + b * 4 + c * 2 + d * 1
  output(hasil)
endprogram
